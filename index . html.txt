            if (activePage) {
                activePage.classList.add('active');
                // Scroll to top of the page content
                window.scrollTo({ top: 0, behavior: 'smooth' });
            }
            // Close mobile menu if open
            mobileMenu.classList.add('hidden');
        }

        // Add click event listeners to navigation links
        navLinks.forEach(link => {
            link.addEventListener('click', (e) => {
                e.preventDefault(); // Prevent default anchor link behavior
                const pageId = link.getAttribute('data-page');
                showPage(pageId);
            });
        });

        // Function to display user messages (instead of alert())
        function alertUserMessage(elementId, message, type) {
            const messageElement = document.getElementById(elementId);
            if (messageElement) {
                messageElement.textContent = message;
                messageElement.classList.remove('hidden', 'bg-red-100', 'text-red-700', 'bg-green-100', 'text-green-700');
                if (type === 'success') {
                    messageElement.classList.add('bg-green-100', 'text-green-700');
                } else if (type === 'error') {
                    messageElement.classList.add('bg-red-100', 'text-red-700');
                }
                setTimeout(() => {
                    messageElement.classList.add('hidden');
                }, 5000); // Hide after 5 seconds
            }
        }

        // --- Tales Page Filtering (Example Static Filter) ---
        const talesCategoryFilter = document.getElementById('tales-category-filter');
        const talesTypeFilter = document.getElementById('tales-type-filter');
        const talesSearchInput = document.getElementById('tales-search');
        const talesGrid = document.getElementById('tales-grid');
        const allStoryCards = talesGrid ? talesGrid.querySelectorAll('.story-card') : [];

        function filterTales() {
            const selectedCategory = talesCategoryFilter.value;
            const selectedType = talesTypeFilter.value;
            const searchTerm = talesSearchInput.value.toLowerCase();

            allStoryCards.forEach(card => {
                const cardCategory = card.getAttribute('data-category');
                const cardType = card.getAttribute('data-type');
                const cardText = card.textContent.toLowerCase();

                const categoryMatch = selectedCategory === 'all' || cardCategory === selectedCategory;
                const typeMatch = selectedType === 'all' || cardType === selectedType;
                const searchMatch = cardText.includes(searchTerm);

                if (categoryMatch && typeMatch && searchMatch) {
                    card.style.display = 'block'; // Show card
                } else {
                    card.style.display = 'none'; // Hide card
                }
            });
        }

        if (talesCategoryFilter) talesCategoryFilter.addEventListener('change', filterTales);
        if (talesTypeFilter) talesTypeFilter.addEventListener('change', filterTales);
        if (talesSearchInput) talesSearchInput.addEventListener('keyup', filterTales);

        // Initial page load: show homepage
        document.addEventListener('DOMContentLoaded', () => {
            showPage('homepage');
            filterTales(); // Apply initial filters to tales page
